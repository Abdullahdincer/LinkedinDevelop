def get_correct_experience_data(record, pagedListComponent, experience_data):
    if pagedListComponent:
        for element in experience_data.get("included", []):
            if element.get("entityUrn") == pagedListComponent:
                record['companyName'] = record['jobTitle']
                elements = element.get("components", {}).get("elements", [])
                if elements:
                    # Güncel unvan çekimi (titleV2 -> text -> text)
                    entity_comp = elements[0].get("components", {}).get("entityComponent", {})
                    title_text_obj = entity_comp.get("titleV2", {}).get("text", {})
                    if isinstance(title_text_obj, dict):
                        record['jobTitle'] = title_text_obj.get("text", "")
                
                # Konum güncelleme
                for component in elements:
                    meta = component.get("components", {}).get("entityComponent", {}).get("metadata", {})
                    if meta and meta.get("text"):
                        record['location'] = meta.get("text", "")
                        break
    return record

def get_experience_datas(profile_urns, follower_number):
    profile_urls = [f'https://www.linkedin.com/voyager/api/graphql?variables=(profileUrn:{urllib.parse.quote(profile_urn)},sectionType:experience,locale:en_US)&queryId=voyagerIdentityDashProfileComponents.a62d9c6739ad5a19fdf61591073dec32' for profile_urn in profile_urns]

    experience_datas = fetch_multiple_data(profile_urls, follower_number)

    if experience_datas:
        records = []
    
        for url, experience_data in experience_datas.items():
            if not (experience_data and "included" in experience_data):
                continue

            included = experience_data["included"]
            componentCount = sum(1 for item in included if "components" in item and item["components"])

            # Mevcut rolü bulmak için döngüleri optimize ediyoruz
            if componentCount > 1:
                for component in included:
                    elements = component.get("components", {}).get("elements", [])
                    if not elements:
                        continue
                    
                    i = 0
                    flag = 0
                    pagedListComponent = ""
                    for element in elements:
                        entity_comp = element.get("components", {}).get("entityComponent")
                        if entity_comp:
                            sub_comp = entity_comp.get("subComponents")
                            if sub_comp and isinstance(sub_comp, dict):
                                comps = sub_comp.get("components", [])
                                if comps and isinstance(comps, list) and len(comps) > 0:
                                    inner_comp = comps[0].get("components", {})
                                    if isinstance(inner_comp, dict) and "*pagedListComponent" in inner_comp:
                                        pagedListComponent = inner_comp.get("*pagedListComponent", "")
                                        flag = 1
                                        break
                        i += 1
                    
                    if flag == 1:
                        record = extract_experience_details(elements[0], experience_data, 1)
                        record = get_correct_experience_data(record, pagedListComponent, experience_data)
                    else:
                        record = extract_experience_details(elements[0], experience_data, 0)
                    
                    record["profileUrl"] = url
                    records.append(record)
                    break # Bu profil için en güncel işi aldık, çıkıyoruz.

            elif componentCount == 1:
                elements = included[0].get("components", {}).get("elements", [])
                if elements:
                    record = extract_experience_details(elements[0], experience_data, 0)
                    record["profileUrl"] = url
                    records.append(record)

        return records
    return []
